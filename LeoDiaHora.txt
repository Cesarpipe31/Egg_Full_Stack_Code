public static void main(String[] args) {
        Scanner leer = new Scanner(System.in);
        System.out.println("Ingrese un tiempo ( en minutos)");
        int min = leer.nextInt();
        System.out.println("La cantidad de minutos ingresados corresponden a: ");
        convertir(min);
    }

    public static void convertir(int min) {
        DecimalFormat form1 = new DecimalFormat("#");
        double dias = min / 1440;
        double hs = Math.floor((min % 1440) / 60);
        System.out.println(form1.format(dias) + " dias, " + form1.format(hs) + " horas.");
    }
}
